74   # around the term unless it's a bare-word. Escape internal quotes
 
75   # if needed. 
76   def solr_param_quote(val, options = {})
 
77     options[:quote] ||= '"'
 
78     unless val =~ /^[a-zA-Z$_\-\^]+$/
 
79       val = options[:quote] +
 
80         # Yes, we need crazy escaping here, to deal with regexp esc too!
 
81         val.gsub("'", "\\\\\'").gsub('"', "\\\\\"") + 
82         options[:quote]
 
83     end
 
84     return val
 
85   end

89   # The CatalogController #index action uses this.
 
90   # Solr parameters can come from a number of places. From lowest
 
91   # precedence to highest:
 
92   #  1. General defaults in blacklight config (are trumped by)
 
93   #  2. defaults for the particular search field identified by  params[:search_field] (are trumped by) 
94   #  3. certain parameters directly on input HTTP query params 
95   #     * not just any parameter is grabbed willy nilly, only certain ones are allowed by HTTP input)
 
96   #     * for legacy reasons, qt in http query does not over-ride qt in search field definition default. 
97   #  4.  extra parameters passed in as argument.
 
98   #
 
99   # spellcheck.q will be supplied with the [:q] value unless specifically
 
100   # specified otherwise. 
101   #
 
102   # Incoming parameter :f is mapped to :fq solr parameter.
 
103   def solr_search_params(user_params = params || {})
 
104     solr_parameters = {}
 
105     solr_search_params_logic.each do |method_name|
 
106       send(method_name, solr_parameters, user_params)
 
107     end
 
108 
109     return solr_parameters
 
110   end

404   # if config is 'true' will look up from Solr @response if available. If

409   def facet_limit_for(facet_field)
 
410     limits_hash = facet_limit_hash
 
411     return nil if limits_hash.blank?
